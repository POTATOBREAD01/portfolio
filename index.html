function predictUsage(maxTemp, weather, month) {
	var isSummer = (month >= 6 && month <= 8);
	var isWinter = (month === 12 || month === 1 || month === 2);

	if (isWinter && weather === '눈') {
		return { text: '증가 예상 ↑', cssClass: 'increase' };
	}
	if (isSummer && maxTemp >= 28) {
	    return { text: '증가 예상 ↑', cssClass: 'increase' };
	}
	if (isWinter && maxTemp <= 5) {
	    return { text: '증가 예상 ↑', cssClass: 'increase' };
	}
	if (weather === '비') {
	    if (isSummer) {
	        return { text: '감소 예상 ↓', cssClass: 'decrease' };
	    }
	    return { text: '유지 예상 →', cssClass: 'neutral' };
	}
	if (maxTemp <= 10) {
	    return { text: '감소 예상 ↓', cssClass: 'decrease' };
	}
	if (maxTemp >= 30) {
	return { text: '증가 예상 ↑', cssClass: 'increase' };
	}

	return { text: '유지 예상 →', cssClass: 'neutral' };
}

  • 폴리곤 데이터가 방대하기 때문에, 거리 기반으로 필요한 폴리곤만 선별하여 불러와 연산하는 A* 알고리즘 사용
  • 연산량 최적화를 위해 대략적인 경로를 연산 후 각 구간을 나누어 정밀 경로를 연산하도록 계층적 경로 탐색 기법 사용

# A* 알고리즘
def astar(start, end, grid_size):
    start_cell = snap_to_grid(start, grid_size)
    end_cell = snap_to_grid(end, grid_size)

    open_set = []
    heapq.heappush(open_set, (0, start_cell))
    came_from = {}
    g_score = {start_cell: 0}

    directions = [
        (0, grid_size), (0, -grid_size),
        (grid_size, 0), (-grid_size, 0),
        (grid_size, grid_size), (grid_size, -grid_size),
        (-grid_size, grid_size), (-grid_size, -grid_size)
    ]

    while open_set:
        _, current = heapq.heappop(open_set)

        if heuristic(current, end_cell) < grid_size / 2:
            path = [end]
            while current in came_from:
                path.append(current)
                current = came_from[current]
            path.append(start)
            path.reverse()
            return path

        for dx, dy in directions:
            neighbor = (current[0] + dx, current[1] + dy)
            if is_land(neighbor[0], neighbor[1]):
                continue

            tentative_g = g_score[current] + heuristic(current, neighbor)
            if neighbor not in g_score or tentative_g < g_score[neighbor]:
                g_score[neighbor] = tentative_g
                f_score = tentative_g + heuristic(neighbor, end_cell)
                heapq.heappush(open_set, (f_score, neighbor))
                came_from[neighbor] = current

    return None
  • snap_to_grid(격자 기반 처리) : 입력받은 start와 end의 좌표를 격자 셀에 맞춰서 정리
  • open_set(탐색 후보) : heapq를 사용하여 최소 f_score(예상 총 비용) 기준으로 정렬되는 우선순위로 구현
  • g_score : 시작점부터 현재 노드까지의 실제 이동 거리
  • f_score : f_score = g_score + heuristic, 현재 노드를 거쳐 도착점까지 가는 예상 비용
  • came_from : 각 노드가 어디에서 왔는지 기록하여 최종 경로 재구성
  • directions(방향 정의) : 8방향 (상하좌우, 대각선)을 정의해 경로를 유연하게 탐색
  • is_land(장애물 회피) : is_land(x, y) 함수로 육지나 섬 여부를 판단
  • heuristic(휴리스틱 함수) : 현재 노드에서 목적지까지의 직선 거리, 도착 가능성을 예측하는 함수
  • 결과 경로 구성 : 목적지에 도달하면, came_from을 따라 시작점부터 도착점까지 경로를 역추적 후 결과 반환

# 계층적 경로 탐색
def hierarchical_pathfinding(start, end):
    # 1단계: 10km coarse grid
    coarse_grid = 0.1
    coarse_path = astar(start, end, coarse_grid)
    if not coarse_path:
        return None

    # 2단계: 각 구간을 1km fine grid로 보정
    final_path = []
    fine_grid = 0.01

    for i in range(len(coarse_path) - 1):
        seg_start = coarse_path[i]
        seg_end = coarse_path[i + 1]
        fine_path = astar(seg_start, seg_end, fine_grid)
        if not fine_path:
            continue  # 이 구간은 연결 실패시 생략 (선택적으로 처리)
        if final_path:
            fine_path = fine_path[1:]  # 중복 지점 제거
        final_path.extend(fine_path)

    return final_path if final_path else None

    1단계: Coarse Grid 탐색 (대략적인 경로 산출)

  • 목적 : 넓은 영역에서 빠르게 대략적인 경로 탐색을 위해 10km 간격의 큰 격자로 A* 알고리즘 실행
  • 효과 : 장애물(육지, 섬) 회피 여부만 대략적으로 고려한 저해상도 탐색으로 연산량을 크게 줄임
  • 결과 : 시작점과 도착점 사이의 큰 윤곽선을 따르는 대략적인 경로 생성

    • 2단계 Fine Grid 보정 (세부 경로 생성)

  • 1단계에서 얻은 대략적인 경로의 구간을 나누어 각 구간을 1km 간격의 세밀한 격자로 세부 경로 탐색

예상 사용자와 이용 목적

  • 위 그래프를 보면 최근 다른 장애인들에 비해 청각장애인의 신규 등록자 수가 압도적으로 높은 걸 알 수 있다.
  • 고령화, 현대 도시 생활에서의 소음 등 여러 가지 이유로 청각장애인 증가
  • 청각장애인들의 수요는 계속해서 증가하고 있으며, 수화와 관련된 기술 및 서비스에 대한 필요성이 커지고 있다.

프레임 워크 선정

  • MediaPipe로 선정

데이터 수집 및 가공

  • 데이터가 존재하지 않아 데이터를 직접 촬영을 해서 수집
  • 수화 동작을 취한 상태에서 각 관절의 좌표(데이터)를 받아와서 데이터셋에 저장하는 방식으로 데이터를 수집
  • 수화를 자음, 모음으로 나누어서 데이터셋을 수집
  • 수집 해야할 수화 동작에 각각 레이블(제스처 번호) 부여
  • 빨간 선 기준 왼쪽은 각 손가락 관절 포인트에 대한 각도와 위치값, 오른쪽으로는 각 동작을 분류하는 레이블(제스처 번호)
  • 'a' 키를 누를 시 취하고 있는 동작의 좌표를 저장하도록 코딩
  • 해당 코드를 실행하면 컴퓨터에 연결된 웹캠에 코드가 연동되어 실행
  • 웹캠에 손이 인식되면 각 관절의 포인트가 등장
  • 수화 동작을 취한 상태에서 'a'키를 누르면 각 관절의 좌표가 저장
  • 위와 같은 방식으로 각 동작의 데이터를 수집하여 데이터셋을 완성

모델 훈련 및 튜닝

데이터 정규화

  • fit 메서드를 사용하여 angle(각도) 데이터를 정규화

데이터 분할

  • 데이터셋을 angle(각도) 값과 label(제스처 번호)로 나누어 훈련세트와 검증세트에 분할

레이블 인코딩

  • 인코딩을 통해 문자열 데이터를 실수형 데이터로 변환

모델 구축

  • Dense 레이어를 이용해서 모델 복잡도를 올려 과소적합을 방지
  • 활성화 함수로 ‘relu’, ‘softmax’ 사용
  • Dropout을 사용하여 데이터 정규화를 진행

옵티마이저 및 학습률 조정

  • Adam 사용, 학습률 설정
  • 손실 함수로는 “sparse_categorical_crossentropy” 함수 사용

조기 종료

  • 과대적합 방지를 위한 조기 종료를 구현

모델 훈련

  • 에포크 및 배치사이즈, 입력데이터 설정
  • 에포크와 배치사이즈를 적절히 수정하면서 정확도를 높임
  • 손실 0.1, 정확도 0.97로 튜닝 완료
